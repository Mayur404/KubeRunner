% ═══════════════════════════════════════════════
\begin{frame}{Slide 2 — Problem Understanding}
  \begin{columns}[T]
    \column{0.55\textwidth}
      \begin{block}{\faExclamationTriangle\ The Problem}
        Modern Kubernetes clusters are composed of \rw{40–200+ interconnected entities}:
        Pods, ServiceAccounts, Roles, Secrets, Databases.
        While each individual permission may appear benign in isolation,
        attackers routinely chain them to reach sensitive resources via
        \rw{multi-hop lateral movement}.\\[6pt]
        Traditional security tools rely on:
        \begin{itemize}
          \item \rw{\faTimesCircle} Static RBAC spreadsheet reviews
          \item \rw{\faTimesCircle} Point-in-time compliance scans
          \item \rw{\faTimesCircle} No graph-theoretic path analysis
          \item \rw{\faTimesCircle} Zero adversary simulation capability
        \end{itemize}
      \end{block}
    \column{0.44\textwidth}
      \begin{alertblock}{\faSkull\ Real Attack Scenario}
        \footnotesize
        \textbf{An attacker compromises a public-facing nginx pod.}\\
        Alone, it has no database access.\\[4pt]
        But it mounts a ServiceAccount token $\rightarrow$\\
        that token is bound to a ClusterRole $\rightarrow$\\
        that role can \texttt{exec} on nodes $\rightarrow$\\
        node access exposes the kube-apiserver $\rightarrow$\\
        apiserver can configure the production DB.\\[4pt]
        \rw{\textbf{5-hop kill chain. Invisible to static analysis.}}
      \end{alertblock}
      \begin{exampleblock}{\faLightbulb\ Key Insight}
        \footnotesize
        Security analysts need a \gw{graph-theoretic, adversary-aware} view of the cluster —
        not just a list of permissions.
      \end{exampleblock}
  \end{columns}
\end{frame}

\begin{frame}{Slide 3 — Proposed Solution}
  \begin{center}
    {\large \kw{KubeRunner} — \textit{See Your Cluster Like An Attacker Does}}
  \end{center}
  \vspace{-4pt}
  \begin{columns}[T]
    \column{0.5\textwidth}
      \feature{\faRoute\ Dijkstra Kill Chain}{
        Finds the \textbf{least-resistant path} from any entry point to any crown jewel.
        Produces a full kill chain with hop-by-hop CVE annotations.
      }
      \feature{\faBiohazard\ Adversary Simulation Engine}{
        Generates a \textbf{step-by-step attacker narrative} with privilege escalation
        tracking, time estimates per hop, and MITRE ATT\&CK® technique IDs.
      }
      \feature{\faStar\ Security Scorecard (0–100)}{
        Aggregates findings into a single cluster \textbf{grade (A–F)} across 6 weighted
        attack-surface categories.
      }
    \column{0.5\textwidth}
      \feature{\faBraille\ Interactive Graph Visualization}{
        A dark-mode Cytoscape.js HTML interface with animated kill chain walkthrough,
        blast zone highlighting, and click-to-inspect on every node.
      }
      \feature{\faFlask\ What-If Remediation Simulator}{
        Simulate \textbf{removing any node or permission} and instantly see the Before vs After
        reduction in exploitable attack paths — quantified as a percentage.
      }
      \feature{\faClock\ Temporal Posture Tracking}{
        Save cluster snapshots and \textbf{diff them over time} to detect newly introduced
        attack paths after each deployment or RBAC change.
      }
  \end{columns}
\end{frame}

% ═══════════════════════════════════════════════
\begin{frame}{Slide 4 — Technical Approach}
  \begin{columns}[T]
    \column{0.36\textwidth}
      \begin{block}{\faProjectDiagram\ Pipeline}
        \footnotesize
        \begin{enumerate}
          \item \kw{Ingest}: Load JSON cluster state \textit{or} scrape live cluster via \texttt{kubectl}. Enrich nodes with CVE CVSS scores.
          \item \kw{Build}: Construct a \textbf{NetworkX DiGraph}. Each entity is a node; each trust relationship is a directed, weighted edge.
          \item \kw{Analyze}: Run 8 graph algorithms in parallel — BFS, Dijkstra, DFS, Centrality, PageRank, Namespace Audit.
          \item \kw{Report}: Generate 8-section Kill Chain Report (CLI → PDF → JSON → HTML).
          \item \kw{Simulate}: Step through the path from an adversary's perspective.
          \item \kw{Score}: Compute the 0–100 Security Scorecard.
        \end{enumerate}
      \end{block}
    \column{0.62\textwidth}
      \begin{tcolorbox}[colback=k8smid,colframe=k8sslate,arc=4pt,title=\textbf{System Architecture},
                        fonttitle=\small\color{k8scyan}]
        \scriptsize
        \begin{verbatim}
  ┌─────────────┐   ┌────────────┐   ┌─────────────────────┐
  │  INGESTOR   │──▶│   GRAPH    │──▶│     ANALYZER        │
  │             │   │  BUILDER   │   │  BFS  Dijkstra  DFS │
  │ kubectl/JSON│   │  NetworkX  │   │  PageRank Centrality│
  │ + CVE DB   │   │  DiGraph   │   │  Namespace Audit    │
  └─────────────┘   └────────────┘   └──────┬──────────────┘
                                            │
              ┌──────────┬─────────────┬────▼──────────────┐
              │          │             │                     │
          ┌───▼───┐ ┌────▼────┐ ┌─────▼──────┐ ┌──────────▼┐
          │REPORTER│ │SIMULATOR│ │ SCORECARD  │ │VISUALIZER │
          │CLI+PDF │ │Narrative│ │  0-100 A-F │ │HTML+JS    │
          └────────┘ └─────────┘ └────────────┘ └───────────┘
        \end{verbatim}
      \end{tcolorbox}
      \vspace{4pt}
      \begin{tcolorbox}[colback=k8smid,colframe=k8syellow,arc=4pt,boxrule=0.5pt]
        \footnotesize
        \textbf{Key Design Choice:} Edge weights represent \textit{exploitability} (lower = easier to exploit).
        Dijkstra thus finds the \textbf{most dangerous path}, not the physically shortest.
      \end{tcolorbox}
  \end{columns}
\end{frame}
